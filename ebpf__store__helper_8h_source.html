46  ebpf_registry_key_t helper_info_key, _In_ const ebpf_helper_function_prototype_t* helper_info)
47 {
48  __return_type status = _SUCCESS;
49  uint32_t offset;
50  ebpf_registry_key_t helper_function_key = NULL;
51  char serialized_data[sizeof(ebpf_helper_function_prototype_t)] = {0};
52 
53  status = create_registry_key_ansi(helper_info_key, helper_info->name, REG_CREATE_FLAGS, &helper_function_key);
54  if (!IS_SUCCESS(status)) {
55  goto Exit;
56  }
57 
58  // Serialize the helper prototype.
59  offset = 0;
60  memcpy(serialized_data, &helper_info->helper_id, sizeof(helper_info->helper_id));
61  offset += sizeof(helper_info->helper_id);
62 
63  memcpy(serialized_data + offset, &helper_info->return_type, sizeof(helper_info->return_type));
64  offset += sizeof(helper_info->return_type);
65 
66  memcpy(serialized_data + offset, helper_info->arguments, sizeof(helper_info->arguments));
67  offset += sizeof(helper_info->arguments);
68 
69  // Save the helper prototype data.
70  status = write_registry_value_binary(
71  helper_function_key, EBPF_HELPER_DATA_PROTOTYPE, (uint8_t*)&serialized_data[0], offset);
72  if (!IS_SUCCESS(status)) {
73  goto Exit;
74  }
75 
76 Exit:
77  if (helper_function_key) {
78  close_registry_key(helper_function_key);
79  }
